    # Analyze task type distribution
    all_task_types = []
    complexity_counts = {'simple': 0, 'multi_step': 0}
    
    for _, row in task_df.iterrows():
        task_types = row['task_types']
        if task_types and isinstance(task_types, list):
            all_task_types.extend(task_types)
            # Determine complexity
            if len(task_types) == 1:
                complexity_counts['simple'] += 1
            else:
                complexity_counts['multi_step'] += 1
    
    # Task type frequency analysis
    task_type_counts = Counter(all_task_types)
    
    print(f"\nTask Type Distribution:")
    for task_type, count in task_type_counts.most_common():
        percentage = (count / len(all_task_types)) * 100
        print(f"  {task_type}: {count} ({percentage:.1f}%)")
    
    print(f"\nComplexity Distribution:")
    total_queries = sum(complexity_counts.values())
    for complexity, count in complexity_counts.items():
        percentage = (count / total_queries) * 100 if total_queries > 0 else 0
        print(f"  {complexity}: {count} ({percentage:.1f}%)")
    
    # Create visualizations
    fig, ((ax1, ax2), (ax3, ax4)) = plt.subplots(2, 2, figsize=(15, 12))
    
    # Task Type Distribution Pie Chart
    if task_type_counts:
        task_types = list(task_type_counts.keys())
        counts = list(task_type_counts.values())
        colors = plt.cm.Set3(range(len(task_types)))
        
        ax1.pie(counts, labels=task_types, autopct='%1.1f%%', colors=colors)
        ax1.set_title('Task Type Distribution')
    
    # Complexity Distribution
    if complexity_counts:
        complexities = list(complexity_counts.keys())
        complexity_values = list(complexity_counts.values())
        
        ax2.bar(complexities, complexity_values, color=['lightblue', 'lightcoral'])
        ax2.set_title('Query Complexity Distribution')
        ax2.set_ylabel('Number of Queries')
    
    # Task Type Frequency Bar Chart
    if task_type_counts:
        task_types_sorted = [item[0] for item in task_type_counts.most_common()]
        counts_sorted = [item[1] for item in task_type_counts.most_common()]
        
        ax3.bar(range(len(task_types_sorted)), counts_sorted, color='lightgreen')
        ax3.set_title('Task Type Frequency')
        ax3.set_ylabel('Frequency')
        ax3.set_xticks(range(len(task_types_sorted)))
        ax3.set_xticklabels(task_types_sorted, rotation=45, ha='right')
    
    # Query length vs Task complexity
    query_lengths = [len(row['user_input']) if row['user_input'] else 0 for _, row in task_df.iterrows()]

        ax4.set_title('Query Length vs Task Complexity')

            display_query = sample[:100] + "..." if len(sample) > 100 else sample
            print(f"  {i}. {display_query}")

        'task_type_counts': task_type_counts,
        'complexity_counts': complexity_counts,
        'total_queries': len(task_df),
        'task_df': task_df
    }
